class SimpleGoogleTest(LiveServerTestCase):
    """Простой тест для проверки Google.com"""
    
    @classmethod
    def setUpClass(cls):
        """Инициализация драйвера"""
        super().setUpClass()
        from selenium import webdriver
        from selenium.webdriver.chrome.options import Options
        from selenium.webdriver.chrome.service import Service
        
        # Путь к драйверу (в папке выше, где manage.py)
        driver_path = os.path.join(os.path.dirname(os.path.dirname(__file__)), "chromedriver.exe")
        service = Service(executable_path=driver_path)
        options = Options()
        options.add_argument("--start-maximized")
        options.add_argument("--disable-blink-features=AutomationControlled")
        
        cls.driver = webdriver.Chrome(service=service, options=options)
        cls.driver.implicitly_wait(10)
    
    @classmethod
    def tearDownClass(cls):
        """Закрытие драйвера"""
        cls.driver.quit()
        super().tearDownClass()
    
    def test_google_status_code(self):
        """Тест 1: Проверка статуса 200 для Google"""
        response = requests.get("https://www.google.com")
        assert response.status_code == 200
        print(f"✓ Google.com вернул статус: {response.status_code}")
    
    def test_google_page_title(self):
        """Тест 2: Проверка заголовка Google через Selenium"""
        self.driver.get("https://www.google.com")
        assert "Google" in self.driver.title
        print(f"✓ Заголовок страницы: {self.driver.title}")
    
    def test_google_search_field(self):
        """Тест 3: Проверка наличия поля поиска"""
        self.driver.get("https://www.google.com")
        
        search_selectors = [
            (By.NAME, "q"),
            (By.CSS_SELECTOR, "input[name='q']"),
            (By.XPATH, "//input[@type='text']")
        ]
        
        search_field = None
        for by, value in search_selectors:
            try:
                search_field = self.driver.find_element(by, value)
                print(f"✓ Найдено поле поиска по локатору: {by}={value}")
                break
            except:
                continue
        
        assert search_field is not None, "Поле поиска не найдено"
        print("✓ Поле поиска присутствует на странице")

from django.shortcuts import render
from django.http import JsonResponse
from django.views.decorators.csrf import csrf_exempt
import subprocess
import json
from io import StringIO
import sys


def test_runner_dashboard(request):
    """Дашборд для запуска тестов"""
    return render(request, 'web_tests/dashboard.html')


@csrf_exempt
def run_tests_api(request):
    """API для запуска тестов"""
    if request.method == 'POST':
        try:
            data = json.loads(request.body)
            test_name = data.get('test_name', 'web_tests.tests.SimpleGoogleTest')
            
            # Запускаем тесты через subprocess
            result = subprocess.run(
                ['python', 'manage.py', 'test', test_name, '--verbosity=2'],
                capture_output=True,
                text=True,
                cwd='..'  # Путь к корню проекта
            )
            
            return JsonResponse({
                'success': True,
                'stdout': result.stdout,
                'stderr': result.stderr,
                'returncode': result.returncode
            })
        except Exception as e:
            return JsonResponse({'success': False, 'error': str(e)})
    
    return JsonResponse({'success': False, 'error': 'Method not allowed'})
