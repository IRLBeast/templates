@csrf_exempt
@require_http_methods(["POST"])
def run_tests_api(request):
    """API для запуска тестов"""
    try:
        data = json.loads(request.body.decode('utf-8'))
        system_key = data.get('system', '')
        test_class_name = data.get('test_class', '')
        
        if system_key not in SYSTEMS:
            return JsonResponse({'success': False, 'error': 'Неизвестная система'}, status=400)
        
        # Запускаем тесты
        result = run_test_class(test_class_name)
        
        return JsonResponse(result)
        
    except Exception as e:
        traceback_str = traceback.format_exc()
        return JsonResponse({
            'success': False,
            'error': str(e),
            'traceback': traceback_str
        }, status=500)


def run_test_class(test_class_name):
    """Запуск конкретного тестового класса"""
    try:
        # Импортируем модуль с тестами
        from main import tests
        
        # Получаем класс тестов
        if not hasattr(tests, test_class_name):
            return {
                'success': False,
                'error': f'Тестовый класс {test_class_name} не найден'
            }
        
        test_class = getattr(tests, test_class_name)
        
        # Создаем загрузчик тестов
        loader = unittest.TestLoader()
        suite = loader.loadTestsFromTestCase(test_class)
        
        # Запускаем тесты с захватом вывода
        import io
        from django.test.runner import DiscoverRunner
        
        # Создаем временный вывод
        stream = io.StringIO()
        
        # Используем стандартный unittest runner
        runner = unittest.TextTestRunner(stream=stream, verbosity=2)
        result = runner.run(suite)
        
        # Получаем вывод
        output = stream.getvalue()
        
        # Формируем отчет
        return {
            'success': result.wasSuccessful(),
            'tests_run': result.testsRun,
            'failures': len(result.failures),
            'errors': len(result.errors),
            'output': output,
            'details': {
                'failures': [(str(f[0]), str(f[1])) for f in result.failures],
                'errors': [(str(e[0]), str(e[1])) for e in result.errors]
            }
        }
        
    except Exception as e:
        traceback_str = traceback.format_exc()
        return {
            'success': False,
            'error': str(e),
            'traceback': traceback_str
        }


def run_all_system_tests(system_key):
    """Запуск всех тестов системы"""
    system = SYSTEMS.get(system_key, {})
    results = {}
    all_success = True
    
    for test_class_name in system.get('test_classes', []):
        result = run_test_class(test_class_name)
        results[test_class_name] = result
        if not result.get('success', False):
            all_success = False
    
    return {
        'success': all_success,
        'results': results
    }

from django.test import LiveServerTestCase

class BaseSeleniumTest(LiveServerTestCase):
    """
    Базовый класс для всех Selenium тестов
    Содержит общие настройки драйвера и вспомогательные методы
    """
    
    @classmethod
    def setUpClass(cls):
        """Настройка драйвера - общая для всех систем"""
        super().setUpClass()
        
        # Путь к драйверу
        driver_path = os.path.join(
            os.path.dirname(os.path.dirname(os.path.dirname(__file__))), 
            "chromedriver.exe"
        )
        
        if not os.path.exists(driver_path):
            raise FileNotFoundError(f"chromedriver.exe not found at {driver_path}")
        
        service = Service(executable_path=driver_path)
        options = Options()
        
        # Общие опции для всех тестов
        options.add_argument("--start-maximized")
        options.add_argument("--disable-blink-features=AutomationControlled")
        options.add_argument("--disable-web-security")
        options.add_argument("--allow-running-insecure-content")
        options.add_argument("--ignore-certificate-errors")
        options.add_argument("--ignore-ssl-errors=yes")
        options.add_experimental_option("excludeSwitches", ["enable-automation"])
        
        # User-agent для разных систем (можно переопределить в наследниках)
        if hasattr(cls, 'USER_AGENT'):
            options.add_argument(f"user-agent={cls.USER_AGENT}")
        
        cls.driver = webdriver.Chrome(service=service, options=options)
        cls.driver.set_page_load_timeout(30)
        cls.driver.implicitly_wait(10)
        
        # Удаление признаков автоматизации
        cls.driver.execute_cdp_cmd("Page.addScriptToEvaluateOnNewDocument", {
            'source': '''
            delete window.cdc_adoQpoasnfa76pfcZLmcfl_Array;
            delete window.cdc_adoQpoasnfa76pfcZLmcfl_Object;
            delete window.cdc_adoQpoasnfa76pfcZLmcfl_Promise;
            delete window.cdc_adoQpoasnfa76pfcZLmcfl_Symbol;
            '''
        })
        
        cls.wait = WebDriverWait(cls.driver, 15)
    
    @classmethod
    def tearDownClass(cls):
        """Закрытие браузера"""
        print(f"\nClosing browser for {cls.__name__}...")
        cls.driver.quit()
        super().tearDownClass()

def run_system_tests(module_path, specific_test_class=None):
    """Запуск тестов для конкретной системы"""
    try:
        # Импортируем модуль с тестами
        test_module = importlib.import_module(module_path)
        
        # Загружаем тесты
        import unittest
        from django.test import TestCase
        
        suite = unittest.TestSuite()
        
        if specific_test_class:
            # Запускаем конкретный тестовый класс
            test_class = getattr(test_module, specific_test_class)
            suite.addTests(unittest.TestLoader().loadTestsFromTestCase(test_class))
        else:
            # Запускаем все тесты в модуле
            for name in dir(test_module):
                obj = getattr(test_module, name)
                if isinstance(obj, type) and issubclass(obj, TestCase):
                    suite.addTests(unittest.TestLoader().loadTestsFromTestCase(obj))
        
        # Запускаем тесты
        from django.test.utils import get_runner
        from django.conf import settings
        
        runner = get_runner(settings)()
        result = runner.run(suite)
        
        # Формируем отчет
        return {
            'success': result.wasSuccessful(),
            'tests_run': result.testsRun,
            'failures': len(result.failures),
            'errors': len(result.errors),
            'details': {
                'failures': [(str(f[0]), str(f[1])) for f in result.failures],
                'errors': [(str(e[0]), str(e[1])) for e in result.errors]
            }
        }
        
    except Exception as e:
        return {
            'success': False,
            'error': str(e),
            'traceback': traceback.format_exc()
        }


def run_all_systems_tests():
    """Запуск тестов всех систем"""
    results = {}
    all_success = True
    
    for system_key, system in SYSTEMS.items():
        if system_key != 'all':
            print(f"\n=== Запуск тестов: {system['name']} ===")
            result = run_system_tests(system['module_path'])
            results[system_key] = result
            if not result.get('success', False):
                all_success = False
    
    return {
        'success': all_success,
        'systems_results': results
    }

async function runTestClass(system, testClass) {
    if (currentRunning) { alert('Тест уже выполняется'); return; }
    
    currentRunning = testClass;
    updateStatus(testClass, 'running', 'Выполняется...');
    document.getElementById('output').innerHTML = `🔄 Запуск ${testClass}...<div class="loading"></div>`;
    
    try {
        const response = await fetch('/run-tests/', {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({ system: system, test_class: testClass })
        });
        const result = await response.json();
        
        if (result.success) {
            updateStatus(testClass, 'passed', `Пройден (${result.tests_run} тестов)`);
            document.getElementById('output').innerHTML = `<span class="success">✅ ТЕСТЫ ПРОЙДЕНЫ</span><br>Выполнено: ${result.tests_run}<br>Ошибок: ${result.errors}<br>Падений: ${result.failures}`;
        } else {
            updateStatus(testClass, 'failed', 'Провален');
            document.getElementById('output').innerHTML = `<span class="error">❌ ТЕСТЫ ПРОВАЛЕНЫ</span><br>${result.error || ''}`;
        }
    } catch (error) {
        updateStatus(testClass, 'failed', 'Ошибка');
        document.getElementById('output').innerHTML = `<span class="error">Ошибка: ${error}</span>`;
    } finally { currentRunning = null; }
}
