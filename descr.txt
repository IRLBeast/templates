from django.views.decorators.csrf import csrf_exempt
import subprocess
import json
from io import StringIO
import sys


def test_runner_dashboard(request):
    """Дашборд для запуска тестов"""
    return render(request, 'web_tests/dashboard.html')


@csrf_exempt
def run_tests_api(request):
    """API для запуска тестов"""
    if request.method == 'POST':
        try:
            data = json.loads(request.body)
            test_name = data.get('test_name', 'web_tests.tests.SimpleGoogleTest')
            
            # Запускаем тесты через subprocess
            result = subprocess.run(
                ['python', 'manage.py', 'test', test_name, '--verbosity=2'],
                capture_output=True,
                text=True,
                cwd='..'  # Путь к корню проекта
            )
            
            return JsonResponse({
                'success': True,
                'stdout': result.stdout,
                'stderr': result.stderr,
                'returncode': result.returncode
            })
        except Exception as e:
            return JsonResponse({'success': False, 'error': str(e)})
    
    return JsonResponse({'success': False, 'error': 'Method not allowed'})
