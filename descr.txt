from django.shortcuts import render
from django.http import JsonResponse
from django.views.decorators.csrf import csrf_exempt
from django.views.decorators.http import require_http_methods
import subprocess
import json
import os
import traceback


def test_runner_dashboard(request):
    """Дашборд для запуска тестов"""
    return render(request, 'main/dashboard.html')


@csrf_exempt
@require_http_methods(["POST"])
def run_tests_api(request):
    """API для запуска тестов"""
    try:
        # Пробуем распарсить JSON
        try:
            data = json.loads(request.body.decode('utf-8'))
            test_name = data.get('test_name', 'main.tests.SimpleGoogleTest')
        except json.JSONDecodeError as e:
            return JsonResponse({
                'success': False,
                'error': f'Ошибка парсинга JSON: {str(e)}',
                'received_body': request.body.decode('utf-8')[:200]
            }, status=400)
        
        # Путь к корню проекта (где manage.py)
        project_root = os.path.dirname(os.path.dirname(os.path.abspath(__file__)))
        
        print(f"Project root: {project_root}")
        print(f"Running test: {test_name}")
        
        # Запускаем тесты через subprocess
        result = subprocess.run(
            ['python', 'manage.py', 'test', test_name, '--verbosity=2'],
            capture_output=True,
            text=True,
            cwd=project_root,
            shell=True  # Добавляем shell=True для Windows
        )
        
        return JsonResponse({
            'success': True,
            'stdout': result.stdout,
            'stderr': result.stderr,
            'returncode': result.returncode
        })
        
    except Exception as e:
        traceback_str = traceback.format_exc()
        return JsonResponse({
            'success': False,
            'error': str(e),
            'traceback': traceback_str
        }, status=500)

<script>
async function runTest(testName) {
    const outputDiv = document.getElementById('outputContent');
    outputDiv.innerHTML = '🔄 Запуск теста ' + testName + '...<br>';
    
    try {
        const response = await fetch('/run-tests/', {
            method: 'POST',
            headers: {
                'Content-Type': 'application/json',
            },
            body: JSON.stringify({ test_name: testName })
        });
        
        // Проверяем статус ответа
        if (!response.ok) {
            const text = await response.text();
            outputDiv.innerHTML = `<span class="error">❌ HTTP ошибка ${response.status}: ${response.statusText}</span><br><br>
                                   <strong>Ответ сервера:</strong><br>
                                   <pre>${text.substring(0, 500)}</pre>`;
            return;
        }
        
        // Пробуем распарсить JSON
        let result;
        try {
            result = await response.json();
        } catch (e) {
            const text = await response.text();
            outputDiv.innerHTML = `<span class="error">❌ Ошибка парсинга JSON: ${e.message}</span><br><br>
                                   <strong>Ответ сервера (первые 500 символов):</strong><br>
                                   <pre>${text.substring(0, 500)}</pre>`;
            return;
        }
        
        if (result.success) {
            outputDiv.innerHTML = '<span class="success">✅ Тест выполнен</span><br><br>' + 
                                 '<strong>STDOUT:</strong><br>' + 
                                 '<pre>' + escapeHtml(result.stdout) + '</pre><br>' +
                                 '<strong>STDERR:</strong><br>' +
                                 '<pre>' + escapeHtml(result.stderr) + '</pre><br>' +
                                 '<strong>Return code:</strong> ' + result.returncode;
        } else {
            outputDiv.innerHTML = '<span class="error">❌ Ошибка: ' + escapeHtml(result.error) + '</span><br><br>' +
                                 (result.traceback ? '<pre>' + escapeHtml(result.traceback) + '</pre>' : '');
        }
    } catch (error) {
        outputDiv.innerHTML = '<span class="error">❌ Ошибка запроса: ' + error + '</span><br><br>' +
                             '<strong>Проверьте:</strong><br>' +
                             '1. Запущен ли сервер<br>' +
                             '2. Правильный ли URL: /run-tests/<br>' +
                             '3. Нет ли ошибок в консоли сервера';
    }
}

// Функция для экранирования HTML
function escapeHtml(text) {
    if (!text) return '';
    const div = document.createElement('div');
    div.textContent = text;
    return div.innerHTML;
}
